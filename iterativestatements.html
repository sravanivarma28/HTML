    5. while

    syntax : 
initilization
while(con){
statements
increment/decrement
}

    while example
    
public class while1 {
public static void main(String[] args) {
	int i=0;
    while(i<=10){
	System.out.println(i);
		
i++;
    }}}

    Do while
    

syntax : 
initilization
do{
statements
increment/decrement
}while(con);


    do while example
    

public class dowhile{
public static void main(String[] args) {
			
      int a=3;
	do {
     System.out.println(a);
	++a;
      }while(a<=9);		
	}}

}

public class For {

  for(int i=0;i<=10;i++) {
       System.out.println(i);
